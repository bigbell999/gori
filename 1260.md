#include <bits/stdc++.h>
using namespace std;

int visited_d[1001];
int visited_b[1001];

vector<int> grp[1001];

int v, e, first;

void dfs(int v);
void bfs(int v);

queue<int> q;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    cin >> v >> e >> first;
    
    int a, b;
    for (int i=0; i<e; i++)
    {
        grp[a].push_back(b);
        grp[b].push_back(a);
    }
    
    //sorting
    for (int i=1; i<v+1; i++)
    {
        if (!grp[i].empty())
        {
            sort(grp[i].begin(), grp[i].end());
        }
    }
    
    //dfs
    dfs(first);
    
    for (int i=1; i<v+1; i++) 
    {
        if (!grp[i].empty()) dfs(i);
    }
    
    cout << '\n';
    
    //bfs
    bfs(first);
    
    for (int i=1; i<v+1; i++)
    {
        if (!grp[i].empty()) bfs(i);
    }
    
    return 0;
}

void dfs(int v)
{
    visited_d[v] = 1;
    cout << v << ' ';
    if (!grp[v].empty())
        for (int i: grp[v])
        {
            if (!visited_d[i]) dfs(i);
        }
}

void bfs(int v)
{
    q.push(v); visited_b[v] = 1;
    while(q.empty())
    {
        if (!grp[v].empty())
        {
            int x = q.front(); q.pop();
            cout << x << ' ';
        
            for (int i: grp[x])
            {
                if (!visited_b[i])
                {
                    visited_b[i] = 1;
                    q.push(i);
                    cout << i << ' ';
                }
            }
        }
    }
}
