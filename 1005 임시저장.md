#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t, n, m;
    
    cin >> t;
    
    for (int k=0; k<t; k++)
    {
        //input
        cin >> n >> m;
        
        //건물 건설시간
        vector<int> demand;
        demand.resize(n+1, 0);
        
        //인접리스트
        vector<int> adj[1001];
        
        //건물의 테크단계
        vector<int> tech;
        tech.resize(n+1, 0);
        
        //진입차수.
        vector<int> in_d;
        in_d.resize(n+1, 0);
        
        //같은 테크 건물들의 건설시간.
        vector<int> tech_time[1001];
        
        //pt 초기화.
        pt.resize(n+1);
        for (int i=1; i<=n; i++) pt[i] = i;
        
        //queue for toplogy sorting.
        queue<int> q;
        
        //input
        for (int i=1; i<=n; i++) cin >> demand[i];
    
        for (int i=0; i<m; i++)
        {
            int a, b;
            cin >> a >> b;
        
            adj[a].push_back(b);
            pt[a] = b;
            in_d[b]++;
        }
        
        //지으면 이기는 건물의 번호.
        int build_2_win;
        cin >> build_2_win;
        
        //solve
        for (int i=1; i<=n; i++)
        {
            if (in_d[i] == 0) 
            {
                q.push(i);
                tech[i] = 1;
            }
        }
    
        while( !q.empty() )
        {
            int cur = q.front();
            q.pop();
        
            for (int i=0; i< adj[cur].size(); i++)
            { 
                int nxt = adj[cur][i];
                if (--in_d[nxt] == 0) 
                {
                    q.push(nxt);
                    tech[nxt] = tech[cur] + 1;
                }
            }
        } 
        
        int tech_2_win = tech[build_2_win];
        
        for (int i=1; i<=n; i++) 
        {
            if ( tech[i] < tech_2_win ) tech_time[tech[i]].push_back( demand[i] );
        }
        
        long long res = 0;
        
        for (int i=1; i<tech_2_win; i++)
        {
            int max = 0;
            for (int j=0; j<tech_time[i].size(); j++)
            {
                if ( tech_time[i][j] > max ) max = tech_time[i][j];
            }
            res += max;
        }
        
        res += demand[build_2_win];
        
        cout << res << '\n';
    }
    
    return 0;
}
