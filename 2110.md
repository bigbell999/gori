#include <bits/stdc++.h>
using namespace std;

int n, c;
vector<int> h;

void init()
{
    cin >> n >> c;
    
    for (int i=0; i<n; i++)
    {
        int a; cin >> a;
        h.push_back(a);
    }
    
    sort(h.begin(), h.end());
}

bool check(int gap, int num)
{
    int cur_idx = 0;
    int cur_val = h[cur_idx];
    num--;
    
    while ( num > 0 )
    {
        cur_idx = lower_bound( h.begin(), h.end(), cur_val + gap ) = h.begin();
        if (cur_idx == n) return false;
        cur_val = h[cur_idx];
        num--;
    }
    
    return true;
}

void solve()
{
    init();
    
    int start = 1;
    int end = h[n-1] - h[0] + 1;
    
    while ( start+1 < end )
    {
        int mid = ( start + end ) / 2;
        
        if ( check(mid, c) ) start = mid;
        else end = mid;
    }
    
    cout << start << '\n';
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    solve();
    
    return 0;
}
