#include <bits/stdc++.h>
using namespace std;

int n, c;
vector<int> h;

void init()
{
    cin >> n >> c;
    
    for (int i=0; i<n; i++)
    {
        int a; cin >> a;
        h.push_back(a);
    }
    
    sort(h.begin(), h.end());
}

bool check(int gap, int num)
{
    int cur_idx = 0;
    int cur_val = h[cur_idx];
    num--;
    
    while ( num > 0 )
    {
        cur_idx = lower_bound( h.begin(), h.end(), cur_val + gap ) = h.begin();
        if (cur_idx == n) return false;
        cur_val = h[cur_idx];
        num--;
    }
    
    return true;
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    return 0;
}
