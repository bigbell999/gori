#include <bits/stdc++.h>
using namespace std;

int n, m;
int maze[101][101];
int dist[101][101];
int visited[101][101];
int destroy = 0;
queue<pair<int, int>> q; 

int dx[4] = {0, 1, 0, -1};
int dy[4] = {1, 0, -1, 0};

void init()
{
    cin >> n >> m;
    for (int i=0; i<n; i++)
    {
        string s;
        cin >> s;
        for (int j=0; j<m; j++) maze[i][j] = s[j] - '0';
    }
}

void bfs()
{
    dist[0][0] = 1;
    visited[0][0] = 1;
    q.push(make_pair(0, 0));
    while(!q.empty())
    {
        int y = q.front().first;
        int x = q.front().second;
        q.pop();
        
        int canGo = 0;
        
        for(int i=0; i<4; i++)
        {
            int ny = y + dy[i];
            int nx = x + dx[i];
            
            if ((0 <= nx) && (nx < m) && (0 <= ny) && (ny < n) && !visited[ny][nx])
            {
                if (!maze[ny][nx])
                {
                    q.push(make_pair(ny, nx));
                    dist[ny][nx] = dist[y][x] + 1;
                    visited[ny][nx] = 1;
                }
                
                if (maze[ny][nx] && !destroy)
                {
                    q.push(make_pair(ny, nx));
                    dist[ny][nx] = dist[y][x] + 1;
                    visited[ny][nx] = 1;
                    destroy = 1;
                }
                
                canGo = 1;
            }
        }
        
        if (!canGo) destroy = 0;
    }
}

int printRep()
{
    if (dist[n-1][m-1] == 0) return -1;
    else return dist[n-1][m-1];
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    init();
    
    bfs();
    
    cout << printRep();
    
    return 0;
}
