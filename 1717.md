#include <bits/stdc++.h>
using namespace std;
using pii = pair<int, int>;

int n, m;
vector<pii> v;

int find(int a)
{
    int parent = v[a].first;
    int child = v[a].second;
    if (parent != child) return find(parent);
    else return a;
}


void is_together(int a, int b)
{
    if (a == b) cout << "YES" << '\n';
    else
    {
        int a_root = find(a);
        int b_root = find(b);
        if (a == b) cout << "YES" << '\n';
        else cout << "NO" << '\n';
    }
}


void union_set(int a, int b)
{
    if (a == b) exit;
    else
    {
        int a_root = find(a);
        int b_root = find(b);
        v[a].first = v[b].second;
    }
}


int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    cin >> n >> m;
    
    v.resize(n+2);
    
    for (int i=0; i<=n; i++) v[i] = make_pair(i, i);
  
    for (int i=0; i<m; i++)
    {
        int order, a, b;
        cin >> order >> a >> b;
        
        if (order == 0) union_set(a, b);
        else is_together(a, b);
    }
  
    return 0;
}
