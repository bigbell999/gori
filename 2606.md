##코드

```
#include <bits/stdc++.h>
using namespace std;

const int MAX  = 101;
int visited[MAX];

vector<int> grp[MAX];

void dfs(int t);

int res = 0;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    int n; cin >> n;
    int m; cin >> m;
    
    for (int i=0; i<m; i++)
    {
        int a, b;
        cin >> a >> b;
        grp[a].push_back(b);
        grp[b].push_back(a);
    }
    
    dfs(1);
    
    cout << res-1;
    
    return 0;
}

void dfs(int t)
{
    res++;
    visited[t] = 1;
    for (int i : grp[t])
    {
        if (!visited[i]) dfs(i);
    }
}
```
