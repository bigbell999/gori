## 처음 코드


```
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

vector<ll> order;
vector<ll> memo;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n; cin >> n;
    order.resize(n); memo.resize(n);
    for (int i=0; i<n; i++) cin >> order[i];
    
    sort(order.begin(), order.end());
    
    if (order[0] != 1)
    {
        cout << 1;
        return 0;
    }
    
    memo[0] = order[0];
    
    ll res = 0;
    for (int i=1; i<n; i++)
    {
        if (order[i] > memo[i-1] + 1)
        {
            res = memo[i-1] + 1;
            break;
        }
        memo[i] = order[i] + memo[i-1]; 
    }
    
    cout << res;
    
    return 0;
}
```

근무서면서 공책에 끄적인 추측으로 짠 코드.  
증명은 못하지만 결과적으로 맞는 듯 하다... 



## 오류

```
for (int i=1; i<n; i++)
    {

        {
            res = memo[i-1] + 1;
            break;

```

(모든 수열의 합 + 1) 인 수를 찾을 수 없다는 예외가 발생했다.  
memo[i-1]을 했기에 모든 수열의 합을 뜻하는 memo[n-1]은 구하지 않는다.

## 그리디하게

```
#include <bits/stdc++.h>
using namespace std;

vector<int> arr;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    int n; cin >> n;
    arr.resize(n);
    for (int i=0; i<n; i++) cin >> arr[i];
    sort(arr.begin(), arr.end());
    
    int x = 1; int y = 1;
    int fin = 0; int clr = 1;
    int idx;
    while(fin == 0)
    {
        if (clr) idx = lower_bound(arr.begin(), arr.end(), y) - arr.begin();
        y -= arr[idx];
        
        if (y == 0)
        {
            x++; 
            y = x;
            clr = 1;
            continue;
        }
        
        if (y < 0)
        {
            if (idx == 0) break;
            y += arr[idx];
        }
        
        if (idx == 0) break;
        
        clr = 0;
        idx--;
    }
    
    cout << x;
    
    return 0;
}
```
