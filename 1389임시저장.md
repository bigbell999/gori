#include <bits/stdc++.h>
using namespace std;


vector<int> grp[101];
vector<int> visited[101];
vector<int> dis
queue<int> q;

int v, e;


int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    // 입력
    cin >> v >> e;
    
    for (int i=0; i<e; i++)
    {
        int a, b;
        cin >> a >> b;
        grp[a].pushback(b);
        grp[b].pushback(a);
    }
    
    // bfs
    q.push(1); visited[1] = 1;
    while(!q.empty())
    {
        int x = q.front(); q.pop();
        
        for (int nx: grp[x])
        {
            if(!visited[x])
            {
                q.push(nx); 
                visited[nx] = 1;
            }
        }
    }
    
    
    
    return 0;
}
