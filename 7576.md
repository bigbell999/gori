#include <bits/stdc++.h>
using namespace std;

int n, m;
int t = 0; // 처음 익은 토마토의 개수를 저장하는 변수
vector<int> box[1001];
vector<int> dist[1001];
vector<int> visited[1001];
vector<pair<int, int> v;
queue<pair<int, int> q;

int dx[4] = {0, 1, 0, -1};
int dy[4] = {1, 0, -1, 0};

void init()
{
    cin >> n >> m;
    for (int i=0; i<m; i++)
    {
        for (int j=0; j<n; j++)
        {
            cin >> box[i][j];
            if (box[i][j] == 1)
            {
                v.push_back(make_pair(i, j));
                t++;
            }
        }
    }
}

void bfs(int y, int x)
{
    visited[y][x] = 1; q.push(make_pair(y, x))
    while(!q.empty())
    {
        int x = q.front(); q.pop()
        
        for (int i=0; i<4; i++)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (!visited[ny][nx] && 0 < ny < m && 0 < nx < n && !box[ny][nx])
            {
                q.push(make_pair(ny, nx));
                if (!dist[ny][nx]) dist[ny][nx] = dist[y][x] + 1;
                else dist[ny][nx] = min(dist[ny][nx], dist[y][x] + 1);
            }
        }
    }
    
    for (int i=0; i<n; i++) fill_n(visited[i], m, 0);
}

int check()
{
    int c = 0;
    for (int i=0; i<n; i++)
    {
        for (int j=0; j<m; j++)
        {
            if (!dist[i][j] && box[i][j] == 0 && box[i][j] == 1) c++;
        }
    }
    
    if (c > t;) return 0;
    if (c == t) return 1;
}

int solve()
{
    for (int i=0; i<t; i++)
    {
        int y = v[i].first();
        int x = v[i].second();
        bfs(y, x);
    }
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    return 0;
}
