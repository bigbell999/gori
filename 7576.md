            }
        }

}

    visited[y][x] = 1; q.push(make_pair(y, x))

            if (!visited[ny][nx] && 0 < ny < m && 0 < nx < n && box[ny][nx] == 0)
            {
                dist[ny][nx] = d;
                q.push(make_pair(ny, nx));
            }
        }

}
