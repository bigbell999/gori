#include <bits/stdc++.h>
using namespace std;

int n, m;
int t = 0; // 처음 익은 토마토의 개수를 저장하는 변수
int box[1001][1001];
int dist[1001][1001]; //처음 익은 토마토부터 각 격자점까지의 최소 거리 저장.
int visited[1001][1001];
vector<pair<int, int>> v; // 처음 익은 토마토의 좌표들을 저장하는 vector
queue<pair<int, int>> q; 

int dx[4] = {0, 1, 0, -1};
int dy[4] = {1, 0, -1, 0};

void init()
{
    cin >> m >> n;
    for (int i=0; i<n; i++)
    {
        for (int j=0; j<m; j++)
        {
            cin >> box[i][j];
            if (box[i][j] == 1)
            {
                v.push_back(make_pair(i, j));
                t++;
            }
        }
    }
}

void bfs(int row, int col)
{
    visited[row][col] = 1; q.push(make_pair(row, col));
    while(!q.empty())
    {
        pair<int, int> p = q.front();
        int y = p.first;
        int x = p.second;
        q.pop();
        
        for (int i=0; i<4; i++)
        {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if (-1 < ny < n && -1 < nx < m && !visited[ny][nx] && box[ny][nx] == 0)
            {
                visited[ny][nx] = 1;
                q.push(make_pair(ny, nx));
                if (!dist[ny][nx]) dist[ny][nx] = dist[y][x] + 1;
                else dist[ny][nx] = min(dist[ny][nx], dist[y][x] + 1);
            }
        }
    }
    
    for (int i=0; i<n; i++) fill_n(visited[i], m, 0);
}

int check() // 모든 토마토가 익을 수 있는지 검사. 익을 수 있다면 1, 아니라면 0 반환.
{
    int c = 0;
    for (int i=0; i<n; i++)
    {
        for (int j=0; j<m; j++)
        {
            if ((!dist[i][j] && box[i][j] == 0) ||  box[i][j] == 1) c++;
        }
    }
    
    if (c > t) return 0;
    else return 1;
}

int solve()
{
    int rep = 0;
    
    for (int i=0; i<t; i++)
    {
        int y = v[i].first;
        int x = v[i].second;
        bfs(y, x);
    }
    
    if (!check()) return -1;
    else
    {
        for (int i=0; i<n; i++)
        {
            for (int j=0; j<m; j++)
            {
                if (dist[i][j] > rep) rep = dist[i][j];
            }
        }
    }
    
    return rep;
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    init();
    
    cout << solve();
    
    return 0;
}
