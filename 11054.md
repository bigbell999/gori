## 임시저장

```
#include <bits/stdc++.h>
using namespace std;

int n;

vector<int> d;
vector<int> l;
vector<int> a;
vector<int> b;

void inc(int k)
{
  for (int i=1; i<n-k; i++)
  {
      if (a[k] < a[k+i]) l[k+i] = max(l[k+i], l[k]+1);
  }
}

void dec(int k)
{
    for (int i=1; i<n-k; i++)
    {
        if (a[k] > a[k+i]) l[k] = max(l[k], l[k+i] + 1);
    }
}

void bi(int k)
{
    int max_l = 0;
    int max_d = 0;
    
    for (int i=0; i<k; i++)
    {
        if (max_l < l[i]) max_l = l[i]; 
    }
    for (int i=k+1; i<n; i++)
    {
        if (max_d < d[i]) max_d = l[i];
    }
    
    b[k] = max_l + max_d + 1;
}

int main()
{
    cin >> n;
    d.resize(n); l.resize(n); a.resize(n); b.resize(n);
    for(int i=0; i<n; i++)
    {
        cin >> a[i];
        d[i] = 1; l[i] = 1;
    }
    
    for (int i=0; i<n-1; i++)
    {
        inc(i); dec(i);
    }
    
    b[0] = max(l[0], d[0]);
    b[n-1] = max(l[0], d[0]);
    
    for (int i=1; i<n-1; i++)
    {
        bi(i);
    }
    
    int res = 0;
    for (int i=0; i<n; i++)
    {
        if (res < b[i]) res = b[i];
    }
    
    cout << res;
    
    return 0;
}
```
