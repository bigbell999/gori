가장 긴 증가하는 부분수열(11053), 가장 긴 감소하는 부분수열(11722)을 적절히 조합하면 될거같다.


## 1트 

```
#include <bits/stdc++.h>
using namespace std;

int n;

vector<int> a;
vector<int> d;
vector<int> l;
vector<int> b;

void inc(int k)
{
    for (int i=k+1; i<n; i++)
    {
        if (a[k] < a[i]) l[i] = max(l[i], l[k]+1);
    }
}

void dec(int k)
{
    for (int i=n-1; i>k; i--)
    {
        if (a[k] > a[i]) d[k] = max(d[k], d[i] + 1);
    }
}

void bi(int k)
{
    int max_l = 0;
    int max_d = 0;
    
    for (int i=0; i<k; i++)
    {
        if (max_l < l[i] && a[i] < a[k]) max_l = l[i]; 
    }
    
    for (int i=k+1; i<n; i++)
    {
        if (max_d < d[i] && a[i] < a[k]) max_d = d[i];
    }
    
    b[k] = max_l + max_d + 1;
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    cin >> n;
    d.resize(n); l.resize(n); a.resize(n); b.resize(n);
    for(int i=0; i<n; i++)
    {
        cin >> a[i];
        d[i] = 1; l[i] = 1;
    }
    
    for (int i=0; i<n; i++)
    {
        inc(i); dec(i);
    }
    
    b[0] = d[0];
    b[n-1] = l[0];
    
    for (int i=1; i<n; i++) bi(i);
    
    int res = 0;
    for (int i=0; i<n; i++)
    {
        if (res < b[i]) res = b[i];
    }
    
    cout << res;
    
    return 0;
}
```

각 수열의 i번째 항(a[i])가 증가수열의 끝항일 때의 증가수열의 길이를 l[i],           
a[i]를 첫항으로 가지는 감소수열의 길이를 d[i]라고 정의했다.     

### 1트 실수
1. 감소수열 길이를 계산하는 함수는 n번째 항부터 첫항까지 계산해야 한다. 그러나 반대로 했기에 결과가 제대로 나오지 않음.    
2. bi함수를 만들 필요가 없다. b[i] = l[i] + d[i] - 1로 계산할 수 있다.

## 성공

```
#include <bits/stdc++.h>
using namespace std;

int n;

vector<int> a;
vector<int> d;
vector<int> l;
vector<int> b;

void inc(int k)
{
    for (int i=k+1; i<n; i++)
    {
        if (a[k] < a[i]) l[i] = max(l[i], l[k]+1);
    }
}

void dec(int k)
{
    for (int i=n-1; i>k; i--)
    {
        if (a[k] > a[i]) d[k] = max(d[k], d[i] + 1);
    }
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    cin >> n;
    d.resize(n); l.resize(n); a.resize(n); b.resize(n);
    for(int i=0; i<n; i++)
    {
        cin >> a[i];
        d[i] = 1; l[i] = 1;
    }
    
    for (int i=0; i<n; i++) inc(i);
    
    for (int i=n-1; i>=0; i--) dec(i);
    
    for (int i=0; i<n; i++) b[i] = l[i] + d[i] - 1;
    
    int res = 0;
    for (int i=0; i<n; i++)
    {
        if (res < b[i]) res = b[i];
    }
    
    cout << res;
    
    return 0;
}
```
