#include <bits/stdc++.h>
using namespace std;

int grp[101][101];
int visited[101][101];
int n, m, k;
vector<int> res;
int area = 0;

int xMv[4] = {0, 1, 0, -1};
int yMv[4] = {1, 0, -1, 0};


void dfs(int x, int y)
{
    area++;
    visited[y][x] = 1;
    int nx, ny;
    for (int i=0; i<4; i++)
    {
        nx = x + xMv[i]; ny = y + yMv[i];
        if ( (0 <= nx && nx < n) && (0 <= ny && ny < m) && !visited[ny][nx] ) dfs(nx, ny);
    }
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    cin >> m >> n >> k;
    res.resize(k);
    
    //입력
    for (int i=0; i<k; i++)
    {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        visited[y1][x1] = 1; visited[y2-1][x2-1] = 1;
    }
    
    //dfs
    int nums = 0;
    for (int i=0; i<m; i++)
    {
        for (int j=0; j<n; j++)
        {
            if (!visited[i][j])
            {
                dfs(j, i);
                res[nums] = area;
                nums++;
                area = 0;
            }
            
        }
    }
    
    sort(res.begin(), res.begin() + nums);
    cout << nums << '\n';
    for (int i=0; i<nums; i++) cout << res[i] << ' ';
    
    return 0;
}
